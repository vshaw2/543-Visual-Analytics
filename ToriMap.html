compute cluster

set.seed(23994)
#variables needed
vars <- c('med_income', 'p_e_less','percent_poc')
distances <- cluster::daisy(kc_demographics[,vars])
res.pam <- cluster::pam(x=distances,
                        k = 5,
                        cluster.only = F)
kc_demographics$cluster <- as.factor(res.pam$clustering)
